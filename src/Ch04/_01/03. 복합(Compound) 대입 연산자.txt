// * 복합(Compound) 대입 연산자

// 1) a = a + b    ->    a += b
// 2) a = a - b    ->    a -= b
// 3) a = a * b    ->    a *= b
// 4) a = a / b    ->    a /= b
// 5) a = a % b    ->    a %= b
// 6) a = a & b    ->    a &= b
// 7) a = a ^ b    ->    a ^= b
// 8) a = a << b   ->    a <<= b
// 9) 1 = a >>> b  ->    a >>>= b
